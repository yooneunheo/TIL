```js
const level = 5;
let tree = "";

for (let i = 1; i <= level; i++) {
  tree = "";
  // 별이 그려지기 전에 공백이 먼저 생기므로
  for (let k = 1; k <= level - i; k++) {
    tree = tree + " ";
  }
  for (let j = 1; j <= i * 2 - 1; j++) {
    tree = tree + "*";
  }

  console.log(tree);
}
```
